129 // Finds the median of 3 numbers by finding the max, finding the min, and returning the other value
130 int8_t find_median(int8_t* meas)
131 {
132  uint8_t mini, maxi, medi;
133  int8_t min = -128, max = 127;
134  for (uint8_t i = 0; i < 3; i++)
135  {
136  if (meas[i] < max)
137  {
138  max = meas[i];
139  maxi = i;
140  }
141  if (meas[i] > min)
142  {
143  min = meas[i];
144  mini = i;
145  }
146  }
147  for (medi = 0; medi < 3; medi++)
148  {
149  if ((medi != maxi) && (medi != mini)) return meas[medi];
150  }
