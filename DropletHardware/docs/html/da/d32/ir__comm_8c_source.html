74 void perform_ir_upkeep()
75 {
76  //For now, can only get one message per call to this function.
77  int8_t msg_chan = -1; //This value is negative if we don't have a message, and otherwise holds the value of the chanel which received the message.
78  int8_t start_dir = (int8_t)(rand_byte()%6);
79  for(int8_t loop_dir= start_dir; loop_dir<(start_dir+6); loop_dir++) //This first loop looks for a channel on which we got a good message.
80  {
81  int8_t dir=(loop_dir%6);
82  if(!(ir_rxtx[dir].status&IR_STATUS_COMPLETE_bm)) continue; //ignore a channel if it isn't done yet. IR_STATUS_COMPLETE should only be set when a transmission is complete.
83  if(ir_rxtx[dir].target_ID) if(ir_rxtx[dir].target_ID != get_droplet_id()) continue;//Is this message targeted, to this droplet?
84  if(ir_rxtx[dir].sender_ID==get_droplet_id()) continue; //ignore a message if it is from me! Silly reflections.
85 
86  uint16_t crc = ir_rxtx[dir].sender_ID;
87  for(uint8_t i=0; i<ir_rxtx[dir].data_length; i++) crc = _crc16_update(crc, ir_rxtx[dir].buf[i]); //Calculate CRC of inbound message.
88  if(ir_rxtx[dir].data_crc == crc) //crc check passed, hurray!
89  {
90  msg_chan = dir;
91  break;
92  }
93  }
94  for(int8_t dir=0;dir<6;dir++) //free up the ir buffers for all the channels we don't need.
95  {
96  if(ir_rxtx[dir].status&IR_STATUS_COMPLETE_bm) if(dir!=msg_chan) clear_ir_buffer(dir);
97  }
98  if(msg_chan>=0) //If we got a good message.
99  {
100  if(ir_rxtx[msg_chan].status & IR_STATUS_COMMAND_bm) //If the message is a command.
101  {
102  char msg[ir_rxtx[msg_chan].data_length+1];
103  memcpy(msg, ir_rxtx[msg_chan].buf, ir_rxtx[msg_chan].data_length);
104  msg[ir_rxtx[msg_chan].data_length]='\0';
105  uint8_t cmd_length = ir_rxtx[msg_chan].data_length;
106  clear_ir_buffer(msg_chan);
107  handle_serial_command(msg, cmd_length);
108  }
109  else //Normal message, so add to msg queue
110  {
111  msg_node* new_node = (msg_node*)malloc(sizeof(msg_node));
112  char* tmp = (char*)malloc(ir_rxtx[msg_chan].data_length+1);
113  new_node->msg = tmp;
114  memcpy(new_node->msg, ir_rxtx[msg_chan].buf, ir_rxtx[msg_chan].data_length);
115  new_node->msg[ir_rxtx[msg_chan].data_length]='\0';
116  new_node->arrival_time = ir_rxtx[msg_chan].last_byte;
117  new_node->arrival_dir = msg_chan;
118  new_node->sender_ID = ir_rxtx[msg_chan].sender_ID;
119  new_node->msg_length = ir_rxtx[msg_chan].data_length;
120  new_node->prev = last_ir_msg;
121  last_ir_msg = new_node;
122  clear_ir_buffer(msg_chan);
123  }
124  }
125  schedule_task(1000/IR_UPKEEP_FREQUENCY, perform_ir_upkeep, NULL);
126  //printf("\r\n");
127 }

220 void ir_receive(uint8_t dir)
221 {
222  uint8_t in_byte = channel[dir]->DATA; // Some data just came in
223  //printf("%02X ",in_byte); //Used for debugging - prints raw bytes as we get them.
224 
225  uint32_t now = get_time();
226  if(now-ir_rxtx[dir].last_byte > IR_MSG_TIMEOUT) ir_rxtx[dir].curr_pos = 0;
227  ir_rxtx[dir].last_byte = get_time();
228 
229  switch(ir_rxtx[dir].curr_pos)
230  {
231  case HEADER_POS_MSG_LENGTH: if(in_byte&DATA_LEN_CMD_bm) ir_rxtx[dir].status|=IR_STATUS_COMMAND_bm; ir_rxtx[dir].data_length = in_byte&DATA_LEN_VAL_bm; break;
232  case HEADER_POS_CRC_LOW: ir_rxtx[dir].data_crc = (uint16_t)in_byte; break;
233  case HEADER_POS_CRC_HIGH: ir_rxtx[dir].data_crc |= (((uint16_t)in_byte)<<8); break;
234  case HEADER_POS_SENDER_ID_LOW: ir_rxtx[dir].sender_ID = (uint16_t)in_byte; break;
235  case HEADER_POS_SENDER_ID_HIGH: ir_rxtx[dir].sender_ID |= (((uint16_t)in_byte)<<8); break;
236  case HEADER_POS_TARGET_ID_LOW: ir_rxtx[dir].target_ID = (uint16_t)in_byte; break;
237  case HEADER_POS_TARGET_ID_HIGH: ir_rxtx[dir].target_ID |= (((uint16_t)in_byte)<<8); break;
238  default: ir_rxtx[dir].buf[ir_rxtx[dir].curr_pos-HEADER_LEN] = in_byte;
239  }
240  ir_rxtx[dir].curr_pos++;
241  if(ir_rxtx[dir].curr_pos>=(ir_rxtx[dir].data_length+HEADER_LEN))
242  {
243  ir_rxtx[dir].status |= IR_STATUS_COMPLETE_bm;
244  ir_rxtx[dir].status |= IR_STATUS_BUSY_bm; //mark as busy so we don't overwrite it.
245  channel[dir]->CTRLB &= ~USART_RXEN_bm; //Disable receiving messages on this channel until the message has been processed.
246  //schedule_task(5, print_completed_msg, (void*)dir);
247  }

259  case HEADER_POS_CRC_LOW: next_byte = (uint8_t)(ir_rxtx[dir].data_crc&0xFF); break;
260  case HEADER_POS_CRC_HIGH: next_byte = (uint8_t)((ir_rxtx[dir].data_crc>>8)&0xFF); break;
261  case HEADER_POS_SENDER_ID_LOW: next_byte = (uint8_t)(ir_rxtx[dir].sender_ID&0xFF); break;
262  case HEADER_POS_SENDER_ID_HIGH: next_byte = (uint8_t)((ir_rxtx[dir].sender_ID>>8)&0xFF); break;
263  case HEADER_POS_TARGET_ID_LOW: next_byte = (uint8_t)(ir_rxtx[dir].target_ID&0xFF); break;

281 void ir_transmit_complete(uint8_t dir)
282 {
283  // this code is being executed because a TXCIF interrupt was executed, see pg. 305 AU manual:
284  // TXCIF: Transmit Complete Interrupt Flag
285  // This flag is set when the entire frame in the transmit shift register has been shifted out and there
286  // are no new data in the transmit buffer (DATA). TXCIF is automatically cleared when the transmit
287  // complete interrupt vector is executed. The flag can also be cleared by writing a one to its bit location.
288  // To do: this suggests we don't really need to do this a byte at a time? Need to check.
289  // Calling this code signals the end of the transmit process
290  //printf("\r\n");
291  TCF2.CTRLB &= ~ir_carrier_bm[dir]; //Turn off the carrier wave.
292 
293  ir_rxtx[dir].status = 0;
294  ir_rxtx[dir].data_length = 0;
295  ir_rxtx[dir].curr_pos = 0;
296  ir_rxtx[dir].target_ID = 0;
297  ir_rxtx[dir].sender_ID = 0;
298  channel[dir]->STATUS |= USART_TXCIF_bm; // writing a 1 to this bit manually clears the TXCIF flag
299  channel[dir]->CTRLB |= USART_RXEN_bm; // this enables receive on the USART
300 }
