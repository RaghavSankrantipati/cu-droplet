34 uint8_t move_steps(uint8_t direction, uint16_t num_steps)
35 {
36  if(is_moving()) return 0;
37  motor_status = MOTOR_STATUS_ON | (direction & MOTOR_STATUS_DIRECTION);
38 
39  uint16_t mot_durs[3]; //This is how long we want each motor to be on for.
40  int8_t mot_dirs[3]; //This is what direction we want each motor to spin in. 1: CCW, -1: CW, 0: No spin.
41  uint16_t total_time = 0; //This is the total length of a step, and will be the period of the PWM generation.
42 
43  int8_t sign_flip;
44  for(uint8_t mot=0 ; mot<3 ; mot++)
45  {
46  if(motor_adjusts[direction][mot]==0)
47  {
48  mot_durs[mot] = 0;
49  mot_dirs[mot] = 0;
50  continue;
51  }
52  else
53  {
54  mot_durs[mot] = 32*motor_on_time + abs(motor_adjusts[direction][mot]);
55  mot_dirs[mot] = ((((motor_adjusts[direction][mot]>>15)&0x1)*-2)+1);
56  total_time += mot_durs[mot] + 32*motor_off_time;
57  }
58  }
59  //printf("Moving in dir: %hhu for %hu steps. Mot_durs: {%hu, %hu, %hu}. Total_time: %hu.\r\n",direction, num_steps, mot_durs[0], mot_durs[1], mot_durs[2], total_time);
60 
61  TCC0.PER = TCC1.PER = TCD0.PER = total_time;
62  TCC0.CCA = TCC0.CCB = mot_durs[0]; //motor 0
63  TCC1.CCA = TCC1.CCB = mot_durs[1]; //motor 1
64  TCD0.CCA = TCD0.CCB = mot_durs[2]; //motor 2
65 
66  uint16_t current_offset = 0;
67 
68  for(uint8_t mot=0 ; mot<3 ; mot++) //This loops sets up the offsets correctly, so that (for example) the 30 ms that motor 1 is on won't start until 40ms after motor 0 turns off.
69  {
70  if(mot_durs[mot]==0) continue;
71  switch(mot)
72  {
73  case 0: TCC0.CNT = ((total_time - current_offset)%total_time); break;
74  case 1: TCC1.CNT = ((total_time - current_offset)%total_time); break;
75  case 2: TCD0.CNT = ((total_time - current_offset)%total_time); break;
76  }
77  current_offset += mot_durs[mot] + 32*motor_off_time;//If we left the motor on for longer to compensate, we should wait a little longer before starting again.
78  }
79  //printf("Offsets are: (%hu, %hu, %hu)\r\n",TCC0.CNT, TCC1.CNT, TCD0.CNT);
80  if(current_offset != total_time) printf("ERROR (I think): current_offset: %hu and total_time: %hu not equal!\r\n", current_offset, total_time);
81  for(uint8_t mot=0 ; mot<3 ; mot++) //Now we just need to tell the motors to go!
82  {
83  if(mot_dirs[mot]<0) motor_backward(mot);
84  else if(mot_dirs[mot]>0) motor_forward(mot);
85  }
86  uint32_t total_movement_duration = ((uint32_t)total_time)*((uint32_t)num_steps)/32;
87  //printf("Total duration: %u ms.\r\n\n",total_movement_duration);
