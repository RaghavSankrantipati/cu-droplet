29 /* This function checks the overall parity of 24 bit codeword cw.
30 If parity is even, 0 is returned, else 1. */
31 {
32  unsigned char p;
33 
34  /* XOR the bytes of the codeword */
35  p=*(unsigned char*)&cw;
36  p^=*((unsigned char*)&cw+1);
37  p^=*((unsigned char*)&cw+2);
38 
39  /* XOR the halves of the intermediate result */
40  p=p ^ (p>>4);

42  p=p ^ (p>>1);

45  return(p & 1);

133 The format of the returned longint is
134 [parity(1),checkbits(11),data(12)]. */
135 {

152 uint32_t correct(uint32_t cw, int8_t *errs)
153 /* This function corrects Golay [23,12] codeword cw, returning the
154 corrected codeword. This function will produce the corrected codeword
155 for three or fewer errors. It will produce some other valid Golay
156 codeword for four or more errors, possibly not the intended
157 one. *errs is set to the number of bit errors corrected. */
158 {
159  uint8_t w; /* current syndrome limit weight, 2 or 3 */
160  uint32_t mask; /* mask for bit flipping */
161  int8_t i,j; /* index */
162  uint32_t s, /* calculated syndrome */
163  cwsaver; /* saves initial value of cw */
164 
165  cwsaver=cw; /* save */
166  *errs=0;
167  w=3; /* initial syndrome weight threshold */
168  j=-1; /* -1 = no trial bit flipping on first pass */
169  mask=1;
170  while (j<23) /* flip each trial bit */
171  {
172  if (j != -1) /* toggle a trial bit */
173  {
174  if (j>0) /* restore last trial bit */
175  {
176  cw=cwsaver ^ mask;
177  mask+=mask; /* point to next bit */
178  }
179  cw=cwsaver ^ mask; /* flip next trial bit */
180  w=2; /* lower the threshold while bit diddling */
181  }
182 
183  s=syndrome(cw); /* look for errors */
184  if (s) /* errors exist */
185  {
186  for (i=0; i<23; i++) /* check syndrome of each cyclic shift */
187  {
188  if ((*errs=weight(s)) <= w) /* syndrome matches error pattern */
189  {
190  cw=cw ^ s; /* remove errors */
191  cw=rotate_right(cw,i); /* unrotate data */
192  //*errs |= IR_RX_STATUS_ECCERR_bm;
193  return(s=cw);
194  }
195  else
196  {
197  cw=rotate_left(cw,1); /* rotate to next pattern */
198  s=syndrome(cw); /* calc new syndrome */
199  }
200  }
201  j++; /* toggle next trial bit */
202  }
203  else
204  return(cw); /* return corrected codeword */
205  }
206 
207  return(cwsaver); /* return original if no corrections */
208 } /* correct */
